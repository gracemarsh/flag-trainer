/**
 * Hook to get the next review date for a specific flag
 * @param flagCode The country code to get the next review date for
 */
export function useNextReviewDate(flagCode: string): string | null {
  const [nextReviewDate, setNextReviewDate] = useState<string | null>(null)
 
  useEffect(() => {
    // Make sure storage is initialized
    initializeStorage()
 
    // Get the flag's progress
    const progress = getAllProgressData()[flagCode]
 
    if (progress) {
      setNextReviewDate(progress.nextReviewDate)
    }
  }, [flagCode])
 
  return nextReviewDate
}
 
/**
 * Hook to manage and monitor synchronization status
 * @param autoInitialize Whether to automatically initialize sync on mount
 */
export function useSyncStatus(autoInitialize = true) {
  const [syncState, setSyncState] = useState<SyncState>(getSyncState())
 
  // Function to refresh the state from the sync module
  const refreshState = useCallback(() => {
    setSyncState(getSyncState())
  }, [])
 
  // Initialize sync on mount
  useEffect(() => {
    if (autoInitialize) {
      initializeSync()
      refreshState()
    }
 
    // Set up periodic state refresh
    const intervalId = setInterval(refreshState, 1000)
 
    // Clean up on unmount
    return () => {
      clearInterval(intervalId)
      if (autoInitialize) {
        cleanupSync()
      }
    }
  }, [autoInitialize, refreshState])
 
  // Function to manually trigger synchronization
  const sync = useCallback(async () => {
    await forceSynchronization()
    refreshState()
  }, [refreshState])
 
  return {
    ...syncState,

  }
